## 一、产品概述

### 1.1 一句话定位

上传简历 → AI 深度诊断 → 多轮模拟面试 → 量化评估报告。帮助求职者发现简历漏洞并练习面试。

### 1.2 核心价值

| 对比维度 | 直接丢给 ChatGPT | AlgoMentor |
|---------|-----------------|------------|
| 给什么 | 完整面试答案 | 分步追问，用户主动回答 |
| 节奏 | 一次性输出 | 4步递进，每步一个焦点 |
| 诊断 | 无针对性 | 基于简历内容标注存疑点 |
| 反馈 | 无 | 量化评分 + 逐点反馈 + 改进建议 |

### 1.3 适用场景

- 求职者准备面试前自查简历漏洞
- 模拟技术面试追问环节
- 简历修改后验证改进效果

### 2.2 技术栈

## 三、完整交互流程

### 3.1 流程总览

```
首页（简历上传）
  │
  ▼
步骤1：简历解析确认（AI 结构化提取 + 用户修正确认）
  │
  ▼

  │
  ▼

  │
  ▼

```

└──────────────────────────────────────────┘
```

进度点可点击回看已完成步骤，未完成不可跳过。

**交互细节**：
- 拖拽到区域高亮边框
- 上传后显示文件名 + 解析进度条
- 解析完成自动跳转步骤1
- 解析失败显示错误提示 + 重试按钮

**后端处理**：
- `POST /api/sessions` 接收文件
- 文件四重校验（扩展名/MIME/魔数/大小）
- 返回 `{session_id, task_id}`
- Celery 异步解析，前端轮询 `GET /api/sessions/{id}/status`

  const [isConnected, setIsConnected] = useState(false);
  const messageQueue = useRef<ClientMessage[]>([]);
  
  const connect = useCallback(() => {
    const socket = new WebSocket(`ws://localhost:8000/ws/interview/${sessionId}`);
    
    socket.onopen = () => {
      setIsConnected(true);
      // 发送缓存消息
      messageQueue.current.forEach(msg => socket.send(JSON.stringify(msg)));
      messageQueue.current = [];
    };
    
    socket.onclose = () => {
      setIsConnected(false);
      // 自动重连（最多5次）
      if (reconnectAttempts < 5) {
        setTimeout(connect, 1000 * Math.min(reconnectAttempts, 5));
        reconnectAttempts++;
      }
    };
    
    socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      handleServerMessage(data);
    };
    
    setWs(socket);
  }, [sessionId]);
  
  const send = (message: ClientMessage) => {
    if (ws?.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify(message));
    } else {
      messageQueue.current.push(message);
    }
  };
  
  useEffect(() => {
    connect();
    return () => ws?.close();
  }, [connect]);
  
  return { send, isConnected };
}
```

### 5.3 LangGraph 状态机

```python
# agent/state.py
from typing import TypedDict, Literal

class AgentState(TypedDict):
    session_id: str
    resume_extract: dict               # 简历解析结果
    doubt_points: list[dict]           # 所有存疑点
    selected_point_ids: list[str]      # 用户勾选的存疑点
    current_point_id: str | None       # 当前追问的存疑点
    current_round: int                 # 当前追问轮次
    messages: list[dict]               # 对话历史
    point_states: dict                 # {point_id: pending|active|resolved|skipped|failed}
    decision: str                      # follow_up | next | report
    error_count: int                   # 连续异常计数
    user_skipped: bool                 # 用户是否主动跳过
    fact_alignment: str | None         # aligned | contradiction | unverifiable
```

```python
# agent/graph.py
from langgraph.graph import StateGraph, END

graph = StateGraph(AgentState)

# 定义节点
graph.add_node("question", question_node)      # 生成追问
graph.add_node("collect", collect_node)         # 收集回答
graph.add_node("evaluate", evaluate_node)       # 评估决策
graph.add_node("report", report_node)           # 生成报告

# 定义边
graph.set_entry_point("question")
graph.add_edge("question", "collect")
graph.add_edge("collect", "evaluate")

# 条件分支
def route_after_evaluate(state: AgentState) -> str:
    if state["decision"] == "follow_up":
        return "question"
    elif state["decision"] == "next":
        return "question"
    else:
        return "report"

graph.add_conditional_edges("evaluate", route_after_evaluate, {
    "question": "question",
    "report": "report"
})

graph.add_edge("report", END)

app = graph.compile()
```

### 5.4 追问防死循环

```python
# agent/rules.py
class InterviewRules:
    MAX_FOLLOW_UP = 3          # 单点最大追问轮次
    MAX_ERROR_COUNT = 3        # 连续异常熔断阈值
    LLM_TIMEOUT = 15           # LLM 超时（秒）
    MAX_TOKENS_PER_SESSION = 50000  # 单会话 Token 上限
    
    @classmethod
    def should_force_switch(cls, state: AgentState) -> tuple[bool, str]:
        if state["current_round"] > cls.MAX_FOLLOW_UP:
            return True, f"超过最大追问轮次({cls.MAX_FOLLOW_UP}轮)"
        if state.get("user_skipped"):
            return True, "用户主动跳过"
        if state["error_count"] >= cls.MAX_ERROR_COUNT:
            return True, f"连续异常({cls.MAX_ERROR_COUNT}次)，触发熔断"
        return False, ""
```

def build_interview_prompt(state: AgentState) -> str:
    parts = []
    
    # 简历关键字段（约 200 tokens）
    parts.append(f"候选人技能：{state['resume_extract']['skills']}")
    parts.append(f"当前追问的简历原文：{state['current_point_source']}")

    parts.append("最近对话：" + "\n".join(

        for m in recent
    ))
    
    # 已确认的能力（约 100 tokens）
    confirmed = [k for k, v in state["point_states"].items() if v == "resolved"]
    parts.append(f"已确认的技能点：{confirmed}")
    
    return "\n\n".join(parts)

### 6.2 Pydantic 校验模型

```python
# models/diagnose.py
from pydantic import BaseModel, field_validator
from typing import Literal

class DoubtPoint(BaseModel):
    id: str
    priority: Literal["high", "medium", "low"]
    source_text: str
    reason: str
    probe_questions: list[str]
    
    @field_validator('probe_questions')
    @classmethod
    def check_questions(cls, v):
        if not v or len(v) == 0:
            raise ValueError("probe_questions 不能为空")
        if len(v) > 5:
            raise ValueError("最多5个追问问题")
        return v
    
    @field_validator('source_text')
    @classmethod
    def check_source(cls, v):
        if len(v) < 5:
            raise ValueError("原文引用过短")
        dangerous = ["忽略", "系统提示", "system prompt"]
        for kw in dangerous:
            if kw.lower() in v.lower():
                raise ValueError(f"疑似注入内容: {kw}")
        return v

class DiagnoseResult(BaseModel):
    overall_score: float
    doubt_points: list[DoubtPoint]
    
    @field_validator('doubt_points')
    @classmethod
    def validate_points(cls, v):
        seen = set()
        for p in v:
            if p.source_text in seen:
                raise ValueError(f"重复引用: {p.source_text[:50]}")
            seen.add(p.source_text)
        return v
```

## 七、安全方案

### 7.1 文件上传安全

```python
# utils/security/file_upload.py
import uuid, magic
from pathlib import Path

ALLOWED_EXTENSIONS = {".pdf", ".docx", ".txt"}
ALLOWED_MIME = {"application/pdf", "application/vnd.openxmlformats-officedocument.wordprocessingml.document", "text/plain"}
MAX_FILE_SIZE = 10 * 1024 * 1024

async def validate_and_save(file: UploadFile) -> tuple[str, str]:
    # 1. 扩展名
    ext = Path(file.filename).suffix.lower()
    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(400, "不支持的文件类型")
    
    # 2. 大小
    contents = await file.read()
    if len(contents) > MAX_FILE_SIZE:
        raise HTTPException(400, "文件过大")
    
    # 3. 魔数
    detected = magic.from_buffer(contents[:2048], mime=True)
    if detected not in ALLOWED_MIME:
        raise HTTPException(400, "文件类型不匹配")
    
    # 4. 随机重命名 + 二级目录
    file_id = str(uuid.uuid4())
    save_dir = Path("uploads") / file_id[:2]
    save_dir.mkdir(parents=True, exist_ok=True)
    save_path = save_dir / f"{file_id}{ext}"
    
    with open(save_path, "wb") as f:
        f.write(contents)
    
    return file_id, str(save_path)
```
